碱基反向互补
basic function
sub reverse_complement {
    my $seq = shift;
    my $revcom_seq = reverse ($seq);    #reverse the DNA seq
    $revcom_seq =~ tr/ACGTacgt/TGCAtgca/;    #complement the reversed DNA seq
    return $revcom_seq;
}
support for all of the IUPAC nucleotide codes (e.g. S = strong = G OR C) and uracil
sub reverse_complement_IUPAC {
    my $seq = shift;
    my $revcom_seq = reverse ($seq);    #reverse the DNA seq
    $revcom_seq =~ tr/ABCDGHMNRSTUVWXYabcdghmnrstuvwxy/TVGHCDKNYSAABWXRtvghcdknysaabwxr/;    #complement the reversed DNA seq
    return $revcom_seq;
}
